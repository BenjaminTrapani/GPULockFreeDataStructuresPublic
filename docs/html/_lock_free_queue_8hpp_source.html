13 /***
14  * Lock-free concurrent queue supporting many concurrent readers and writers. This is an implementation
15  * of the lock-free queue described here: https://www.research.ibm.com/people/m/michael/podc-1996.pdf
16  * The implementation relies on device-side allocations, which may be slow. To avoid leaking memory / get correct
17  * results, there must not be any ref count collisions in a log2(queueNodeAlignment) bit counter.
18  * This might be unlikely depending on how the queue is accessed. A tradeoff between minimum required queue node size
19  * can be adjusted by modifying queueNodeAlignment in the AlignmentConfigs namespace below.
20  */
21 namespace LockFreeQueueGPU {

23  namespace AlignmentConfigs {
24  constexpr unsigned int queueNodeAlignment = 16;
25  //Must always equal log2(queueNodeAlignment). Unfortunately no constexpr log2
26  constexpr unsigned int numCountBits = 4;
27  constexpr unsigned int ptrMask = queueNodeAlignment - 1;
28  }

39  __device__ void enqueue(const T &value) {
40  //Perform device-side memory allocation (likely slow)
41  QueueNode *newNode = new QueueNode(value);
42  QueueNodePointer tail;
43  while (true) {
44  tail = _tail;
45  const QueueNodePointer next = tail.ptr()->next;
46  if (tail == _tail) {
47  if (next.ptr() == nullptr) {
48  if (tail.ptr()->next.performAtomicCAS(next, QueueNodePointer(newNode, next.count() + 1))) {
49  break;
50  }
51  } else {
52  _tail.performAtomicCAS(tail, QueueNodePointer(next.ptr(), tail.count() + 1));
53  }
54  }
55  }
56  _tail.performAtomicCAS(tail, QueueNodePointer(newNode, tail.count() + 1));
57  }
58 
59  __device__ bool dequeue(T &result) {
60  QueueNodePointer head;
61  while (true) {
62  head = _head;
63  const QueueNodePointer tail = _tail;
64  const QueueNodePointer next = head.ptr()->next;
65  if (head == _head) {
66  if (head.ptr() == tail.ptr()) {
67  if (next.ptr() == nullptr) {
68  return false;
69  }
70  _tail.performAtomicCAS(tail, QueueNodePointer(next.ptr(), tail.count() + 1));
71  } else {
72  result = next.ptr()->value;
73  if (_head.performAtomicCAS(head, QueueNodePointer(next.ptr(), head.count() + 1))) {
74  break;
75  }
76  }
77  }
78  }
79  delete head.ptr();
80  return true;
81  }

114  __device__ bool performAtomicCAS(const QueueNodePointer &expectedNode, const QueueNodePointer &newNode) {
115  Pointer_t *pointerThis = reinterpret_cast<Pointer_t *>(this);
116  //Reinterpret cast to pointers here because the size of QueueNodePointer is not known at this point
117  // during compilation, so the compiler cannot allocate the right size for a full copy on the stack.
118  Pointer_t const *pointerExpectedNode = reinterpret_cast<Pointer_t const*>(&expectedNode);
119  Pointer_t const *pointerNewNode = reinterpret_cast<Pointer_t const*>(&newNode);
120  const Pointer_t updatedVal = atomicCAS(pointerThis, *pointerExpectedNode, *pointerNewNode);
121  return updatedVal == *pointerExpectedNode;
122  }
